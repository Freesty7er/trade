Процедура ЗаполнитьЗначениямиПоУмолчанию(обновитьЗначения=Ложь) Экспорт
	
	тдЗначенияПоУмолчанию = ПолучитьМакет("ЗначенияПоУмолчанию");
	
	Для номерСтроки = 2 По тдЗначенияПоУмолчанию.ВысотаТаблицы Цикл
		
		адресКод = СтрШаблон("R%1C1", Формат(номерСтроки, "ЧГ="));
		адресНаименование = СтрШаблон("R%1C2", Формат(номерСтроки, "ЧГ="));
		адресПоУмолчанию = СтрШаблон("R%1C4", Формат(номерСтроки, "ЧГ="));
		адресТипПО = СтрШаблон("R%1C5", Формат(номерСтроки, "ЧГ="));
		
		сКод = тдЗначенияПоУмолчанию.Область(адресКод).Текст;
		сНаименование = тдЗначенияПоУмолчанию.Область(адресНаименование).Текст;
		сПоУмолчанию = тдЗначенияПоУмолчанию.Область(адресПоУмолчанию).Текст;
		сТипПО = тдЗначенияПоУмолчанию.Область(адресТипПО).Текст;
		
		спрВидыправДоступаСсылка = Справочники.Р_ВидыПравДоступаОператоров.НайтиПоКоду(сКод);
		
		Если спрВидыправДоступаСсылка.Пустая() Тогда
			спрВидыправДоступа = Справочники.Р_ВидыПравДоступаОператоров.СоздатьЭлемент();
		ИначеЕсли обновитьЗначения Тогда
			спрВидыправДоступа = спрВидыправДоступаСсылка.ПолучитьОбъект();
		Иначе
			Продолжить;
		КонецЕсли;
			
		спрВидыправДоступа.Код 			= сКод;
		спрВидыправДоступа.Наименование = сНаименование;
		спрВидыправДоступа.ПоУмолчанию	= ?(ВРег(сПоУмолчанию)="НЕТ", Ложь, Истина);
		
		Если Не ПустаяСтрока(сТипПО) Тогда
			спрВидыправДоступа.ТипПО = Перечисления.Р_ТипыПОПОС[сТипПО];
		КонецЕсли;
			
		спрВидыправДоступа.Записать();
			
		
		
	КонецЦикла;
	
КонецПроцедуры	// ЗаполнитьЗначениямиПоУмолчанию()

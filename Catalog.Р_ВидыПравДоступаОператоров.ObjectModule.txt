
#Область ОбработчикиСобытий

Процедура ПередЗаписью(отказ)
	
	Если ОбменДанными.Загрузка Тогда
		Возврат;
	КонецЕсли;
	
	Если ПустаяСтрока(Код) Тогда
		
		ТекстСообщения = НСтр("ru = 'Запись невозможна без заполнения поля ""Код"".'");
		ПроверкаДанныхКлиентСервер.СообщитьОбОшибке(отказ, текстСообщения, ЭтотОбъект, "Код");
		
	КонецЕсли;
	
	Если ПустаяСтрока(Наименование) Тогда
		
		ТекстСообщения = НСтр("ru = 'Запись невозможна без заполнения поля ""Наименование"".'");
		ПроверкаДанныхКлиентСервер.СообщитьОбОшибке(отказ, текстСообщения, ЭтотОбъект, "Наименование");
		
	КонецЕсли;
	
КонецПроцедуры	// ПередЗаписью()
